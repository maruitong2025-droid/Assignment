clear
clc

L = 256;
nx = 513;
dx = L/(nx-1);
x = linspace(0,L,nx);

D = 1.6e-6;
dt = 43200;

r = D*dt/dx^2;

t_days = [10 100 1000 10000];
t_sec = t_days*24*3600;

nt = ceil(t_sec(end)/dt);

T = -40*ones(1,nx);
T(1) = 2;
T(end) = -40;

T_store = zeros(4,nx);

for n = 1:nt
    T_new = T;
    for i = 2:nx-1
        T_new(i) = T(i) + r*(T(i+1)-2*T(i)+T(i-1));
    end
    T_new(1) = 2;
    T_new(end) = -40;
    T = T_new;

    t_now = n*dt;

    for k = 1:4
        if T_store(k,1)==0 && t_now>=t_sec(k)
            T_store(k,:) = T;
        end
    end
end

figure
hold on
plot(x,T_store(1,:),'LineWidth',1.5)
plot(x,T_store(2,:),'LineWidth',1.5)
plot(x,T_store(3,:),'LineWidth',1.5)
plot(x,T_store(4,:),'LineWidth',1.5)
legend('10 d','100 d','1000 d','10000 d')
xlabel('x (m)')
ylabel('Temperature (°C)')
grid on

L = 256;
nx = 513;
nt = 50;
nF = 250;

dx = L/(nx-1);

t_days = [10, 100, 1000, 10000];

for k = 1:length(t_days)
    
    t_final = t_days(k)*86400;
    
    T = analytical_solution(nx, nt, nF, L, t_final, 0);
    
    temp = T(:,end);
    
    melted_nodes = find(temp > 0);
    
    if isempty(melted_nodes)
        length_melted = 0;
    else
        length_melted = max(melted_nodes)*dx;
    end
    
    fprintf('t = %d days, melted length = %.2f m\n', t_days(k), length_melted);
    
end

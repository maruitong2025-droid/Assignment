clear
clc

L = 256;
D = 1.6e-6;
nx = 513;
x = linspace(0,L,nx);

nF = 250;
days = [10 100 1000 10000];

for k = 1:length(days)
    
    t = days(k)*24*3600;
    
    T = -42*x/L + 2;
    
    for n = 1:nF
        T = T + (-84/(n*pi)) * sin(n*pi*x/L) * exp(-D*(n*pi/L)^2*t);
    end
    
    id = find(T > 0);
    
    if isempty(id)
        melt_length = 0;
    else
        melt_length = x(max(id));
    end
    
    fprintf('At t = %5d days, melted length = %.1f m\n', days(k), melt_length)
end

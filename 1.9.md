clear
clc

nx = 513;
nt = 2;
nF = 250;
tend = 10*24*3600;
plotflag = 1;

L = 256;
D = 1.6e-6;

x = linspace(0,L,nx);
t = linspace(0,tend,nt);

T = zeros(nx,nt);

for j = 1:nt
    
    tj = t(j);
    
    y = -42*x/L + 2;
    
    for n = 1:nF
        y = y + (-84/(n*pi))*sin(n*pi*x/L)*exp(-D*(n*pi/L)^2*tj);
    end
    
    T(:,j) = y';
    
    if plotflag == 1
        plot(x,y)
        xlabel('x (m)')
        ylabel('Temp (deg C)')
        title(['t = ',num2str(tj/3600),' h'])
        grid on
        drawnow
    end
    
end

end

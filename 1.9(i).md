L = 256;
nx = 513;
nt = 2;
t_final = 12*3600; % 12 hours

nF_list = [10, 100, 250, 500, 1000];

figure
hold on

for k = 1:length(nF_list)
    
    nF = nF_list(k);
    
    T = analytical_solution(nx, nt, nF, L, t_final, 0);
    
    x = linspace(0, L, nx);
    
    plot(x, T(:,end), 'LineWidth', 1.5)
    
end

xlabel('x (m)')
ylabel('Temperature (°C)')
legend('nF=10','nF=100','nF=250','nF=500','nF=1000')
title('Analytical solution at t = 12 hours')
grid on

clear
clc
close all

L = 256;
D = 1.6e-6;
nx = 513;
x = linspace(0,L,nx);

nFlist = [10 100 250 500 1000];
t = 12*3600;

figure
hold on

for k = 1:length(nFlist)
    nF = nFlist(k);

    T = -42*x/L + 2;

    for n = 1:nF
        T = T + (-84/(n*pi)) * sin(n*pi*x/L) * exp(-D*(n*pi/L)^2*t);
    end

    plot(x,T,'LineWidth',1.2,'DisplayName',['nF=',num2str(nF)])
end

xlabel('x (m)')
ylabel('Temp (deg C)')
title('Analytical solution at t = 12 h')
legend
grid on
axis([0 20 -50 2])

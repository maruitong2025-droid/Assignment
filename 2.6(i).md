clear
clc

L = 256;
nx_list = [65 129 257 513];
D = 1.6e-6;
dt = 43200;

figure
hold on

for k = 1:length(nx_list)

    nx = nx_list(k);
    dx = L/(nx-1);
    x = linspace(0,L,nx);
    r = D*dt/dx^2;

    T = -40*ones(1,nx);
    T(1)=2; T(end)=-40;

    t_target = 10*24*3600;
    nt = ceil(t_target/dt);

    for n = 1:nt
        T_new = T;
        for i = 2:nx-1
            T_new(i) = T(i) + r*(T(i+1)-2*T(i)+T(i-1));
        end
        T_new(1)=2; T_new(end)=-40;
        T = T_new;
    end

    plot(x,T,'LineWidth',1.2)
end

xlabel('x (m)')
ylabel('Temperature (°C)')
legend('65','129','257','513')
axis([0 10 -40 -35])
grid on

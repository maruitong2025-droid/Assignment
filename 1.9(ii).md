clear
clc
close all

L = 256;
D = 1.6e-6;
nx = 513;
x = linspace(0,L,nx);
nF = 250;

days = [10 100 1000 10000];

figure
hold on

for k = 1:length(days)
    t = days(k)*24*3600;

    T = -42*x/L + 2;

    for n = 1:nF
        T = T + (-84/(n*pi)) * sin(n*pi*x/L) * exp(-D*(n*pi/L)^2*t);
    end

    plot(x,T,'LineWidth',1.5,'DisplayName',['t = ',num2str(days(k)),' d'])
end

xlabel('x (m)')
ylabel('Temperature (^oC)')
title('Temperature profile along glacier')
legend
grid on
axis([0 256 -45 5])

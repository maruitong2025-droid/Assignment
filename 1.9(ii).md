L = 256;
nx = 513;
nt = 50;
nF = 250;

t_days = [10, 100, 1000, 10000];

figure
hold on

for k = 1:length(t_days)
    
    t_final = t_days(k)*86400;
    
    T = analytical_solution(nx, nt, nF, L, t_final, 0);
    
    x = linspace(0, L, nx);
    
    plot(x, T(:,end), 'LineWidth', 2)
    
end

xlabel('x (m)')
ylabel('Temperature (°C)')
legend('10 d','100 d','1000 d','10000 d')
title('Temperature profiles over time')
grid on

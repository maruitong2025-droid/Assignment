clear
clc

L = 256;
nx = 513;
dx = L/(nx-1);
x = linspace(0,L,nx);

D = 1.6e-6;
dt = 80000;   
r = D*dt/dx^2;

nt = 200;

T = -40*ones(1,nx);
T(1) = 2;
T(end) = -40;

for n = 1:nt
    T_new = T;
    for i = 2:nx-1
        T_new(i) = T(i) + r*(T(i+1)-2*T(i)+T(i-1));
    end
    T_new(1) = 2;
    T_new(end) = -40;
    T = T_new;
end

figure
plot(x,T,'LineWidth',1.5)
xlabel('x (m)')
ylabel('Temperature (°C)')
title('Unstable numerical solution')
grid on
